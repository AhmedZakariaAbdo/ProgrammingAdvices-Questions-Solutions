#include <iostream>
#include <iomanip>
#include <vector>
#include <fstream>
#include <string>
#include <ctime>
using namespace std;

// Define a structure to hold bank client information
struct sClient {
    string AccountNumber;
    string PinCode;
    string Name;
    string Phone;
    double AccountBalance;
};

// Function to split a string by a delimiter and return a vector of substrings
vector<string> SplitString(string s1, string Delim) {
    vector<string> Vstring;
    short pos = 0;
    string sWord;

    // Loop to extract substrings until the delimiter is no longer found
    while ((pos = s1.find(Delim)) != std::string::npos) {
        sWord = s1.substr(0, pos);

        if (sWord != "") {
            Vstring.push_back(sWord); // Add non-empty substring to the result
        }

        s1.erase(0, pos + Delim.length()); // Remove the processed part from the string
    }

    if (s1 != "") {
        Vstring.push_back(s1); // Add the last part
    }

    return Vstring;
}

// Function to convert a delimited string line to a client record
sClient ConvertLineTorecord(string stline, string Sperator = "#//#") {
    sClient Client;
    vector<string> vClientData;

    vClientData = SplitString(stline, Sperator); // Split line into individual fields

    // Assign each part to the appropriate field in the struct
    Client.AccountNumber = vClientData[0];
    Client.PinCode = vClientData[1];
    Client.Name = vClientData[2];
    Client.Phone = vClientData[3];
    Client.AccountBalance = stod(vClientData[4]); // Convert string to double

    return Client;
}

// Function to print a client record to the console
void PrintClientRecord(sClient Client) {
    cout << "\n\nThe Following is The extracted client record: \n";
    cout << "\nAccount Number: " << Client.AccountNumber << endl;
    cout << "\nAccount PinCode: " << Client.PinCode << endl;
    cout << "\nAccount Name: " << Client.Name << endl;
    cout << "\nAccount Phone: " << Client.Phone << endl;
    cout << "\nAccount Balance: " << Client.AccountBalance << endl;
}

int main() {
    srand((unsigned)time(NULL)); // Seed the random number generator (not used here, but good practice)

    // Example line representing a client's data in a delimited format
    string stline = "A150#//#1234#//#Ahmed Zakaria#//#01097478961#//#10000.0000";
    
    cout << "\nLine Record Is\n";
    cout << stline;

    // Convert the line to a client record and print it
    sClient client = ConvertLineTorecord(stline);
    PrintClientRecord(client);

    return 0;
}

vector<string> SplitString(string s1, string Delim) {

    while ((pos = s1.find(Delim)) != std::string::npos) {

sClient ConvertLinetoRecord(string line, string sperator = "#//#") {

    // Assign each part of the line to the corresponding struct field

using namespace std;

// File name that stores all clients data
const string ClientDatafile = "Clients.txt";

// Structure that represents one client record
struct sClients
{
    string AccountNumber;   // Unique account number
    string PinCode;         // Client PIN code
    string AccountName;     // Client full name
    string Phone;           // Client phone number
    string Email;           // Client email address
    double AccountBalance;  // Client account balance
};

// Splits a string based on a given delimiter
// Example: "A#//#B#//#C" → ["A", "B", "C"]
vector<string> SplitString(string S1, string Delim)
{
    vector<string> Vstring;
    short pos = 0;
    string sword;

    // Loop until no delimiter is found
    while ((pos = S1.find(Delim)) != string::npos)
    {
        // Take substring before the delimiter
        sword = S1.substr(0, pos);

        // Avoid adding empty strings
        if (sword != "")
        {
            Vstring.push_back(sword);
        }

        // Remove processed part + delimiter
        S1.erase(0, pos + Delim.length());
    }

    // Add the remaining part after the last delimiter
    if (S1 != "")
    {
        Vstring.push_back(S1);
    }

    return Vstring;
}

// Converts one line from the file into a client structure
sClients ConvertLineToRecord(string Line, string Separator = "#//#")
{
    // Split the line into individual fields
    vector<string> Vstring = SplitString(Line, Separator);

    sClients Client;

    // Assign each field to its corresponding struct member
    Client.AccountNumber = Vstring[0];
    Client.PinCode = Vstring[1];
    Client.AccountName = Vstring[2];
    Client.Phone = Vstring[3];
    Client.Email = Vstring[4];
    Client.AccountBalance = stod(Vstring[5]); // Convert string to double

    return Client;
}

// Loads all clients data from file into a vector
vector<sClients> LoadDataFromFile(string FileName)
{
    vector<sClients> vClientsData;
    fstream Myfile;

    // Open file in read mode
    Myfile.open(FileName, ios::in);

    if (Myfile.is_open())
    {
        string Line;
        sClients Client;

        // Read file line by line
        while (getline(Myfile, Line))
        {
            // Convert each line to a client record
            Client = ConvertLineToRecord(Line);

            // Store client in vector
            vClientsData.push_back(Client);
        }

        Myfile.close(); // Close file after reading
    }

    return vClientsData;
}

    cout << "\nPhone          : " << Client.Phone;
    cout << "\nEmail          : " << Client.Email;
    cout << "\nAccount Balance: " << Client.AccountBalance;

        }

    return false; // Client not found

    }
    else
    {
        cout << "\nClient With Account Number (" 
             << AccountNumber

using namespace std;

// File name that stores clients data
const string ClientDatafile = "Clients.txt";

// Structure that represents one client record
struct sClients
{
    string AccountNumber;     // Unique account number
    string PinCode;           // Client PIN code
    string AccountName;       // Client name
    string Phone;             // Client phone number
    string Email;             // Client email
    double AccountBalance;    // Client balance
    bool MarkForUpdate = true; // Flag (currently always true)
};

// Splits a string based on a given delimiter
vector<string> SplitString(string S1, string Delim)
{
    vector<string> Vstring;
    short pos = 0;
    string sword;

    // Loop until delimiter is found
    while ((pos = S1.find(Delim)) != std::string::npos)
    {
        sword = S1.substr(0, pos);

        // Avoid adding empty strings
        if (sword != "")
            Vstring.push_back(sword);

        // Remove processed part from string
        S1.erase(0, pos + Delim.length());
    }

    // Add remaining part
    if (S1 != "")
        Vstring.push_back(S1);

    return Vstring;
}

// Converts a line from file into a client structure
sClients ConvertLineToRecord(string Line, string Separator = "#//#")
{
    vector<string> Vstring = SplitString(Line, Separator);
    sClients Client;

    Client.AccountNumber  = Vstring[0];
    Client.PinCode        = Vstring[1];
    Client.AccountName    = Vstring[2];
    Client.Phone          = Vstring[3];
    Client.Email          = Vstring[4];
    Client.AccountBalance = stod(Vstring[5]); // Convert string to double

    return Client;
}

// Converts client structure into a string to save in file
string ConvertRecordToLine(sClients Client, string Separator = "#//#")
{
    string stClientRecord = "";

    stClientRecord += Client.AccountNumber + Separator;
    stClientRecord += Client.PinCode + Separator;
    stClientRecord += Client.AccountName + Separator;
    stClientRecord += Client.Phone + Separator;
    stClientRecord += Client.Email + Separator;
    stClientRecord += to_string(Client.AccountBalance);

    return stClientRecord;
}

// Loads all clients from file into a vector
vector<sClients> LoadDataFromFile(string FileName)
{
    vector<sClients> vClientsData;
    fstream Myfile;

    // Open file for reading only
    Myfile.open(FileName, ios::in);

    if (Myfile.is_open())
    {
        string Line;
        sClients Client;

        // Read file line by line
        while (getline(Myfile, Line))
        {
            Client = ConvertLineToRecord(Line);
            vClientsData.push_back(Client);
        }

        Myfile.close();
    }

    return vClientsData;
}

    fstream MyFile;
    MyFile.open(FileName, ios::out); // Overwrite file

    string Dataline;

    if (MyFile.is_open())
    {
        // Write all clients to file
        for (sClients C : vClients)
        {
            if (C.MarkForUpdate == true)
            {
                Dataline = ConvertRecordToLine(C);
                MyFile << Dataline << endl;
            }
        }
        MyFile.close();
    }

    return vClients;
}
