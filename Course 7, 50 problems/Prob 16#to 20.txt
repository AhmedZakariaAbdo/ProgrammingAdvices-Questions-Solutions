    bool IsSParse(int Matrix[3][3],int Rows,int Cols){

        short zerocount = 0;
        short nonzero = 0;

        for(short i = 0; i < Cols; i ++){

            for(short j = 0; j < Rows; j++){

                if (Matrix[i][j] == 0){
                    
                    zerocount ++;
                }else {

                    nonzero++;
                }
            }
        }

        return zerocount > nonzero;
    }

        PrintMatrix(Matrix1,3,3);
        
        cout<<"\n------------------"<<endl;

        if(IsSParse(Matrix1,3,3)){
            
            cout<<"\nYes: It's Sparse"<<endl;
        }else{
            cout<<"\nNo: It's not Sparse"<<endl;
        }

    return 0;
}

Problem 17#
Number Exists in Matrix
===================================
Description:
Write A Program to check if a given number Exists in matrix or not
==============================================================================
My Solution:
==============

#include <iostream>
#include <iomanip>
#include <vector>
#include <fstream>
#include <string>
#include <ctime>
using namespace std;

// Function to generate a random number between From and To (inclusive)
int RandomNumber(int From, int To) {
    int RandNum = rand() % (To - From + 1) + From;
    return RandNum;
}

// Function to fill a matrix with random numbers (only 0 or 1 in this case)
void FillMatrixWithRandomNumbers(int Arr[3][3], int Rows, int Cols) {
    for (short i = 0; i < Cols; i++) {
        for (short j = 0; j < Rows; j++) {
            Arr[i][j] = RandomNumber(0, 1);
        }
    }
}

// Function to print the contents of a matrix
void PrintMatrix(int arr[3][3], int Rows, int cols) {
    for (short i = 0; i < cols; i++) {
        for (short j = 0; j < Rows; j++) {
            printf(" %0*d   ", 2, arr[i][j]); // print with 2-digit width
        }
        cout << endl;
    }
}

// Function to search for a specific number in the matrix
bool FindNumberInMatrix(int Matrix[3][3], int Number, int Rows, int Cols) {
    for (short i = 0; i < Cols; i++) {
        for (short j = 0; j < Rows; j++) {
            if (Matrix[i][j] == Number)
                return true; // return true as soon as the number is found
        }
    }
    return false; // number not found in the matrix
}

int main() {

    // Seed the random number generator with the current time
    srand((unsigned)time(NULL));

    // Initialize a 3x3 matrix with specific values
    int Matrix1[3][3] = { {5, 0, 0}, {6, 45, 10}, {5, 5, 15} };

    cout << "\nMatrix1: \n";
    PrintMatrix(Matrix1, 3, 3); // Display the matrix

    cout << "\n------------------" << endl;

    int Number = 0;
    cout << "\nPlease Enter The Number You Are Looking For: ";
    cin >> Number;

    // Check if the number exists in the matrix
    if (FindNumberInMatrix(Matrix1, Number, 3, 3)) {
        cout << "\nYes: It Exists" << endl;
    } else {
        cout << "\nNo: It Does Not Exist" << endl;
    }

    return 0;
}

Write A Program to check it the matrix is Palindrom or not

==============

int RandomNumber(int From, int To){

}

    void FillMatrixWithRandomNumbers(int Arr[3][3], int Rows, int Cols){

        for(short i = 0; i < Cols; i ++){

                Arr[i][j] = RandomNumber(0,1);
            }
        }

    void PrintMatrix(int arr[3][3],int Rows, int cols){

        for(short i = 0; i < cols; i++){

            for(short j = 0; j < Rows; j++){

                printf(" %0*d   ",2,arr[i][j]);
               // cout<<setw(3)<< arr[i][j]<<"    ";
            }

        }
    }


    int IsPalindromMatrix(int matrix1[3][3], int Rows, int Cols){

        
        for(short i = 0; i < Rows; i ++){

            for(short j = 0; j < Cols / 2; j++){

                if(matrix1[i][j] != matrix1[i][Cols - 1 - j]){
                    return false;
                }
            }
        }
        return true;
    }


    int main(){

        srand((unsigned)time(NULL));

        int Matrix1[3][3] = {{5,0,0},{6,45,10},{5,9,15}};
        

        cout<<"\nMatrix1: \n";
        PrintMatrix(Matrix1,3,3);
        
        cout<<"\n------------------"<<endl;
    
        if(IsPalindromMatrix(Matrix1,3,3)){
            cout<<"\nYes: Matrix is palindrome"<<endl;

        }else{
            cout<<"\nNo: Matrix Is not Palindrome"<<endl;
        }

    return 0;
}
